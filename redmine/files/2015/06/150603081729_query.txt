select distinct ind.id_incident , (select org.ds_description from sdk_quarters qua , sdk_centre cen , sdk_organization org where qua.nu_fkcentre = cen.id_centre and cen.nu_fkorganization = org.id_organization and qua.id_quarters = ind.nu_fkquarter ) "Organismo" , (select DECODE(inds.lg_is_indic_reached,1,'CORRECTO','INCORRECTO') from sdk_indicator_incident inds where inds.nu_fkindicatordefs = 5000000 and inds.nu_fkincident = ind.id_incident) "Tiempo de Respuesta" , (select inds.nu_evaluationvalue from sdk_indicator_incident inds where inds.nu_fkindicatordefs = 5000000 and inds.nu_fkincident = ind.id_incident) "Tiempo de Respuesta (V)" , (select DECODE(inds.lg_is_indic_reached,1,'CORRECTO','INCORRECTO') from sdk_indicator_incident inds where inds.nu_fkindicatordefs = 5000001 and inds.nu_fkincident = ind.id_incident) "Tiempo de Resolución" , (select inds.nu_evaluationvalue from sdk_indicator_incident inds where inds.nu_fkindicatordefs = 5000001 and inds.nu_fkincident = ind.id_incident) "Tiempo de Resolución (V)" , (select DECODE(inds.lg_is_indic_reached,1,'CORRECTO','INCORRECTO') from sdk_indicator_incident inds where inds.nu_fkindicatordefs = 5000005 and inds.nu_fkincident = ind.id_incident and inds.tx_phase = 'Pendiente Usuario') "Pendiente Usuario" , (select inds.nu_evaluationvalue from sdk_indicator_incident inds where inds.nu_fkindicatordefs = 5000005 and inds.nu_fkincident = ind.id_incident and inds.tx_phase = 'Pendiente Usuario') "Pendiente Usuario (V)" , oper.tx_name as tipo_actuacion , (select ds_description from sdk_prioritycode pc where pc.id_prioritycode = ind.nu_fkcurrent_priority_code) "Prioridad" from sdk_incident ind , sdk_phaseservice pse , sdk_serviceversion sce , sdk_indicator_incident inds , sdk_generalstate gst , sdk_itservice its , sdk_operation oper where (ind.fh_closedate >= :f1 and ind.fh_closedate <= :f2) and ind.nu_fkphaseservice = pse.id_phaseservice and pse.nu_fkserviceversion = sce.id_serviceversion and ind.id_incident = inds.nu_fkincident and gst.id_generalstate = pse.nu_fkgeneralstate and gst.cd_code <> 'REGISTRO' and its.tx_code = 'SIC-AGATADES' and ind.nu_fkitservice = its.id_itservice and ind.NU_FKOPERATION = oper.id_operation order by 1;
