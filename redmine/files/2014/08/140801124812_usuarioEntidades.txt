-- Entidades que administra un usuario
SELECT eg.uid_entidad AS uid_usuario,
  ent.acronimo
FROM usuario us
JOIN entidad_guia eg
ON us.id_usuario = eg.id_entidad_guia
JOIN usuario_admin_entidad uae
ON uae.id_usuario = us.id_usuario
JOIN entidad ent
ON ent.id_entidad = uae.id_entidad
WHERE eg.uid_entidad LIKE 'prueba06.cci.chap';
